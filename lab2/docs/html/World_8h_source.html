224 for (auto &pos : extraEnemies) {
225 int choice = Utils::RandomInt(0,3);

228 if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;

230 if ((nx == npcX && ny == npcY) || (hasMerchant && nx == merchantX && ny == merchantY)) continue;
231 // also avoid player and main enemy
232 if ((nx == playerX && ny == playerY) || (enemyPresent && nx == enemyX && ny == enemyY)) continue;
233 // avoid colliding with other extras
234 bool occupiedByExtra = false; for (const auto &p2 : extraEnemies) { if (&p2 != &pos && p2.first == nx && p2.second == ny) { occupiedByExtra = true; break; } }
235 if (occupiedByExtra) continue;
236 grid[pos.second][pos.first].SetType('.');
237 grid[pos.second][pos.first].SetOccupied(false);
238 pos.first = nx; pos.second = ny;
